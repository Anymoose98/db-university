SELECT YEAR(`enrolment_date`) AS anno_di_nascita, COUNT(*) AS numero_di_iscritti FROM `students` GROUP BY YEAR(`enrolment_date`); SELECT COUNT(*) AS `name`, `office_address` FROM `teachers` GROUP BY `office_address`; SELECT `vote`, AVG(`vote`) FROM `exam_student`; SELECT COUNT(*) AS `numero_corsi`, `department_id` FROM `degrees` GROUP BY `department_id`; SELECT `students`.* FROM `students` JOIN `degrees` ON `students`.`degree_id` = `degrees`.`id` WHERE `degrees`.`name` = 'corso di economia'
